# 재귀가 없는 일반 형태
def factorial_norec(n):
    answer = 1
    for i in range(2, n + 1):
        answer *= i
    return answer
                        

# 재귀 형태
def factorial_rec(n):
    # 종료 조건
    if n <= 1:
        return 1
    return n * factorial_rec(n - 1)

class Stack:
    def __init__(self):
        self.l = []
    def push(self, new_elem):
        self.l.append(new_elem)
    def pop(self):
        popped = self.l[-1]
        del self.l[-1]
        return popped
    def top(self):
        return self.l[-1]
    def is_empty(self):
        return len(self.l) == 0
